      //var == variable == 변수 선언 구문 
      var 변수명; // 전역변수 (같은 문서 내 선언된 이후부터 어디서든 사용가능 )

      변수명; // 전역변수 

      function 함수명(){
        var 변수명; // function 지역변수

        변수명; // 전역변수 

        if(조건식{
          var 변수명; // function의 지역변수
          const 변수명; // if문의 지역변수
          let 변수명; // if문의 지역변수
        })
      }

      </script>

var 변수 형식은 전역 또는 function 지역변수 만을 범위로 갖는다.

변수 선언 방식

    var : 변수 function level scope
    const : 상수 block{} level scope ({}를 벗어나면 접근 불가)
    let : const와 같이 블록레벨이나 변수
    
    +var변수는 변수명 중복이 가능하지만 
    const,let은 변수명 중복이 불가능하다.

    1순위 : const 
    2순의 : let

js의 자료형

    자바스크립트는 변수 선언 시 별도의 자료형을 지정하지 않는다.
    상수인가 변수인가만 지정. 
    -> 변수에 대입되는 값에 의하여 변수의 자료형이 결정된다. 

    -string(문자열) 
      JS는 char자료형이 존재하지 않으므로 길이에 상관하지 않고 모두 문자열.
      ''와 ""로 표시 가능 
    
    - Number (숫자)
      정수, 실수 모두 

    - boolean (논리형)
      ture, false

    - object(객체) 
      Array, List, Set, Map, HTML요소

    - function (함수)
      const testFn = function(){}

      -undefined(정의 되지 않은 변수)
        변수 선언만하고 값 대입x
        ex) let str;
            console.log(str); // -undefined
